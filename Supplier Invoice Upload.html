  <script>

    const form = document.getElementById('invoiceForm');

    const responseDiv = document.getElementById('response');


    form.addEventListener('submit', async (e) => {

      e.preventDefault();


      responseDiv.innerHTML = '';

      const formData = new FormData(form);


      try {

        const res = await fetch('https://YOUR-N8N-DOMAIN/webhook/supplier-bill-upload', {

          method: 'POST',

          body: formData

        });


        const data = await res.json();


        if (data.success) {

          responseDiv.innerHTML = `<div class="success">Invoice uploaded successfully. Bill No: ${data.bill_number}</div>`;

          form.reset();

        } else {

          responseDiv.innerHTML = `<div class="error">${data.message || 'Upload failed'}</div>`;

        }

      } catch (error) {

        responseDiv.innerHTML = '<div class="error">Server error. Please try again later.</div>';

      }

    });

  </script>
